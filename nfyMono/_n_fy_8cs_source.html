217 public void OpenSong(string path)
218 {
219
220 AudioStreamPlayer audSound = getNFyStream();
221
222 File file = new File();
223 Error err = file.Open(path, File.ModeFlags.Read);
224 byte[] b = file.GetBuffer((int)file.GetLen());
225
226
227 if (path.Extension() == "wav")
228 {
229 AudioStreamSample a = new AudioStreamSample();
230 a.Format = AudioStreamSample.FormatEnum.Format16Bits;
231 a.Stereo = true;
232 a.Data = b;
233 SongLength = a.GetLength();
234 audSound.Stream = a;
235
236 }
237
238 else if (path.Extension() == "ogg")
239 {
240 AudioStreamOGGVorbis d = new AudioStreamOGGVorbis();
241 d.Data = b;
242 SongLength = d.GetLength();
243 audSound.Stream = d;
244 }
245
246 else if (path.Extension() == "mp3")
247 {
248 AudioStreamMP3 d = new AudioStreamMP3();
249 d.Data = b;
250 SongLength = d.GetLength();
251 audSound.Stream = d;
252 }
253
254
255
256 audSound.Play();
257 }
